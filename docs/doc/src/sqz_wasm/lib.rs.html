1//! # sqz-wasm
2//!
3//! Browser-compatible WASM build of the sqz compression engine. This is a
4//! self-contained subset of `sqz_engine` compiled to `wasm32-unknown-unknown` —
5//! no tree-sitter, no SQLite, no filesystem. Everything runs in memory.
6//!
7//! ## What's included
8//!
9//! - TOON encoding for JSON (null stripping, array collapsing, tabular encoding)
10//! - Content classification (JSON, code, logs, prose)
11//! - Text compression with phrase abbreviation and word shortening
12//! - Log condensing (repeated lines collapsed)
13//! - Code compression (comment stripping, blank line collapsing)
14//! - In-memory dedup cache (hash-based, no persistence)
15//! - Three compression presets: minimal, default, aggressive
16//!
17//! ## Browser usage
18//!
19//! Build with `wasm-pack`:
20//!
21//! ```text
22//! wasm-pack build sqz-wasm --target web
23//! ```
24//!
25//! Then in JavaScript:
26//!
27//! ```text
28//! import init, { SqzWasm } from './pkg/sqz_wasm.js';
29//!
30//! await init();
31//! const sqz = new SqzWasm("default");
32//!
33//! // Compress JSON — gets TOON-encoded, nulls stripped
34//! const result = sqz.compress('{"name": "Alice", "debug": null}');
35//! console.log(result); // TOON:{name:"Alice"}
36//!
37//! // Estimate token count
38//! const tokens = sqz.estimate_tokens(result);
39//!
40//! // Session export/import for persistence
41//! const ctx = sqz.export_ctx();
42//! sqz.import_ctx(ctx);
43//! ```
44//!
45//! ## Presets
46//!
47//! - `"minimal"` — TOON encoding only, no stripping
48//! - `"default"` — strip nulls + condense + TOON + word abbreviation
49//! - `"aggressive"` — all of default + collapse arrays at 5 items
50

60fn encode_value(v: &serde_json::Value, buf: &mut String) {
61    match v {
62        serde_json::Value::Null => buf.push_str("null"),
63        serde_json::Value::Bool(b) => buf.push_str(if *b { "true" } else { "false" }),
64        serde_json::Value::Number(n) => {
65            buf.push_str(&serde_json::to_string(&serde_json::Value::Number(n.clone()))
66                .unwrap_or_else(|_| n.to_string()));
67        }
68        serde_json::Value::String(s) => encode_string(s, buf),
69        serde_json::Value::Array(arr) => {
70            buf.push('[');
71            for (i, item) in arr.iter().enumerate() {
72                if i > 0 {
73                    buf.push(',');
74                }
75                encode_value(item, buf);
76            }
77            buf.push(']');
78        }
79        serde_json::Value::Object(map) => {
80            buf.push('{');
81            for (i, (k, val)) in map.iter().enumerate() {
82                if i > 0 {
83                    buf.push(',');
84                }
85                if is_simple_key(k) {
86                    buf.push_str(k);
87                } else {
88                    encode_string(k, buf);
89                }
90                buf.push(':');
91                encode_value(val, buf);
92            }
93            buf.push('}');
94        }
95    }
96}

98fn encode_string(s: &str, buf: &mut String) {
99    buf.push('"');
100    for ch in s.chars() {
101        match ch {
102            '"' => buf.push_str("\\\""),
103            '\\' => buf.push_str("\\\\"),
104            '\n' => buf.push_str("\\n"),
105            '\r' => buf.push_str("\\r"),
106            '\t' => buf.push_str("\\t"),
107            c if (c as u32) < 0x20 => {
108                buf.push_str(&format!("\\u{:04x}", c as u32));
109            }
110            c if (c as u32) > 0x7E => {
111                let cp = c as u32;
112                if cp <= 0xFFFF {
113                    buf.push_str(&format!("\\u{:04x}", cp));
114                } else {
115                    let cp = cp - 0x10000;
116                    let high = 0xD800 + (cp >> 10);
117                    let low = 0xDC00 + (cp & 0x3FF);
118                    buf.push_str(&format!("\\u{:04x}\\u{:04x}", high, low));
119                }
120            }
121            c => buf.push(c),
122        }
123    }
124    buf.push('"');
125}

127fn is_simple_key(k: &str) -> bool {
128    if k.is_empty() {
129        return false;
130    }
131    let mut chars = k.chars();
132    let first = chars.next().unwrap();
133    if !first.is_ascii_alphabetic() && first != '_' {
134        return false;
135    }
136    if !chars.all(|c| c.is_ascii_alphanumeric() || c == '_') {
137        return false;
138    }
139    !matches!(k, "true" | "false" | "null")
140}

146/// Recursively remove null-valued fields from JSON objects.
147/// Arrays keep their null elements (positional data).
148fn strip_nulls(value: &mut serde_json::Value) {
149    match value {
150        serde_json::Value::Object(map) => {
151            map.retain(|_, v| !v.is_null());
152            for v in map.values_mut() {
153                strip_nulls(v);
154            }
155        }
156        serde_json::Value::Array(arr) => {
157            for item in arr.iter_mut() {
158                strip_nulls(item);
159            }
160        }
161        _ => {}
162    }
163}

165/// Collapse arrays with more than `max_items` elements.
166/// For uniform arrays (all objects with same keys), use tabular encoding.
167fn collapse_arrays(value: &mut serde_json::Value, max_items: usize) {
168    match value {
169        serde_json::Value::Array(arr) => {
170            for item in arr.iter_mut() {
171                collapse_arrays(item, max_items);
172            }
173            if arr.len() > max_items {
174                // Check for uniform array → tabular encoding
175                if let Some(keys) = detect_uniform_keys(arr) {
176                    let table = encode_table(arr, &keys);
177                    let count = arr.len();
178                    arr.clear();
179                    arr.push(serde_json::Value::String(
180                        format!("[table: {} rows]\n{}", count, table),
181                    ));
182                } else {
183                    let remaining = arr.len() - max_items;
184                    arr.truncate(max_items);
185                    arr.push(serde_json::Value::String(
186                        format!("... and {} more items", remaining),
187                    ));
188                }
189            }
190        }
191        serde_json::Value::Object(map) => {
192            for v in map.values_mut() {
193                collapse_arrays(v, max_items);
194            }
195        }
196        _ => {}
197    }
198}

200fn detect_uniform_keys(arr: &[serde_json::Value]) -> Option<Vec<String>> {
201    if arr.len() < 2 {
202        return None;
203    }
204    let first_keys: Vec<String> = match &arr[0] {
205        serde_json::Value::Object(map) if !map.is_empty() => map.keys().cloned().collect(),
206        _ => return None,
207    };
208    for item in &arr[1..] {
209        match item {
210            serde_json::Value::Object(map) => {
211                if map.len() != first_keys.len() {
212                    return None;
213                }
214                for key in &first_keys {
215                    if !map.contains_key(key) {
216                        return None;
217                    }
218                }
219            }
220            _ => return None,
221        }
222    }
223    Some(first_keys)
224}

226fn encode_table(arr: &[serde_json::Value], keys: &[String]) -> String {
227    let mut lines = Vec::with_capacity(arr.len() + 1);
228    lines.push(keys.join(" | "));
229    for item in arr {
230        if let serde_json::Value::Object(map) = item {
231            let row: Vec<String> = keys
232                .iter()
233                .map(|k| compact_value(map.get(k).unwrap_or(&serde_json::Value::Null)))
234                .collect();
235            lines.push(row.join(" | "));
236        }
237    }
238    lines.join("\n")
239}
240
241fn compact_value(v: &serde_json::Value) -> String {
242    match v {
243        serde_json::Value::Null => "null".to_string(),
244        serde_json::Value::Bool(b) => b.to_string(),
245        serde_json::Value::Number(n) => n.to_string(),
246        serde_json::Value::String(s) => {
247            if s.len() > 50 { format!("{}...", &s[..47]) } else { s.clone() }
248        }
249        serde_json::Value::Array(a) => format!("[{} items]", a.len()),
250        serde_json::Value::Object(m) => format!("{{{} keys}}", m.len()),
251    }
252}

258#[derive(Debug, Clone, Copy, PartialEq, Eq)]
259enum ContentType {
260    Json,
261    Code,
262    Log,
263    Prose,
264}
265
266fn classify(input: &str) -> ContentType {
267    let trimmed = input.trim();
268    if trimmed.starts_with('{') || trimmed.starts_with('[') {
269        if serde_json::from_str::<serde_json::Value>(trimmed).is_ok() {
270            return ContentType::Json;
271        }
272    }
273    let lines: Vec<&str> = trimmed.lines().take(20).collect();
274    let mut code_score = 0;
275    let mut log_score = 0;
276    for line in &lines {
277        let l = line.trim();
278        if l.ends_with('{') || l.ends_with('}') || l.ends_with(';')
279            || l.starts_with("fn ") || l.starts_with("def ")
280            || l.starts_with("class ") || l.starts_with("import ")
281            || l.contains("->") || l.contains("::")
282        {
283            code_score += 1;
284        }
285        if l.contains("[INFO]") || l.contains("[ERROR]") || l.contains("[WARN]")
286            || l.contains("[DEBUG]") || l.starts_with("20")
287        {
288            log_score += 1;
289        }
290    }
291    if code_score > lines.len() / 2 { return ContentType::Code; }
292    if log_score > lines.len() / 3 { return ContentType::Log; }
293    ContentType::Prose
294}

300fn compress_text(input: &str) -> String {
301    let mut result = String::with_capacity(input.len());
302
303    // Pass 1: Normalize whitespace and collapse blank lines
304    let mut prev_blank = false;
305    for line in input.lines() {
306        let trimmed = line.trim();
307        if trimmed.is_empty() {
308            if !prev_blank {
309                result.push('\n');
310                prev_blank = true;
311            }
312            continue;
313        }
314        prev_blank = false;
315        let mut prev_space = false;
316        for ch in trimmed.chars() {
317            if ch == ' ' || ch == '\t' {
318                if !prev_space {
319                    result.push(' ');
320                    prev_space = true;
321                }
322            } else {
323                result.push(ch);
324                prev_space = false;
325            }
326        }
327        result.push('\n');
328    }
329
330    // Pass 2: Phrase abbreviations
331    let result = result
332        .replace("for example", "e.g.")
333        .replace("For example", "E.g.")
334        .replace("that is to say", "i.e.")
335        .replace("in order to", "to")
336        .replace("In order to", "To")
337        .replace("as well as", "&")
338        .replace("such as", "e.g.")
339        .replace("a number of", "several")
340        .replace("A number of", "Several")
341        .replace("due to the fact that", "because")
342        .replace("Due to the fact that", "Because")
343        .replace("in the event that", "if")
344        .replace("In the event that", "If")
345        .replace("at this point in time", "now")
346        .replace("At this point in time", "Now")
347        .replace("it is important to note that ", "")
348        .replace("It is important to note that ", "")
349        .replace("it should be noted that ", "")
350        .replace("It should be noted that ", "")
351        .replace("the fact that ", "")
352        .replace("there is a need to", "must")
353        .replace("There is a need to", "Must")
354        .replace("in terms of", "regarding")
355        .replace("In terms of", "Regarding")
356        .replace("with respect to", "regarding")
357        .replace("With respect to", "Regarding")
358        .replace("on the other hand", "conversely")
359        .replace("On the other hand", "Conversely")
360        .replace("in addition to", "besides")
361        .replace("In addition to", "Besides")
362        .replace("as a result of", "from")
363        .replace("As a result of", "From")
364        .replace("  ", " ");
365
366    // Pass 3: Word abbreviations (same table as the Rust engine)
367    let result = abbreviate_words(&result);
368
369    let result = result.trim_end().to_string();
370    if result.len() < input.len() { result } else { input.to_string() }
371}

373/// Collapse repeated consecutive lines (condense stage).
374fn condense(input: &str, max_repeated: usize) -> String {
375    let mut result = Vec::new();
376    let mut current_line: Option<&str> = None;
377    let mut run_count: usize = 0;
378
379    for line in input.lines() {
380        match current_line {
381            Some(prev) if prev == line => {
382                run_count += 1;
383                if run_count <= max_repeated {
384                    result.push(line);
385                }
386            }
387            _ => {
388                current_line = Some(line);
389                run_count = 1;
390                result.push(line);
391            }
392        }
393    }
394
395    let trailing = input.ends_with('\n');
396    let mut out = result.join("\n");
397    if trailing { out.push('\n'); }
398    out
399}

401/// Compress log output: condense repeated lines, keep errors/warnings.
402fn compress_log(input: &str) -> String {
403    condense(input, 2)
404}

406/// Compress code: strip single-line comments, condense blank lines.
407fn compress_code(input: &str) -> String {
408    let mut result = Vec::new();
409    let mut prev_blank = false;
410    for line in input.lines() {
411        let trimmed = line.trim();
412        // Strip single-line comments (but keep doc comments)
413        if (trimmed.starts_with("//") && !trimmed.starts_with("///"))
414            || (trimmed.starts_with('#') && !trimmed.starts_with("#!") && !trimmed.starts_with("#["))
415        {
416            continue;
417        }
418        if trimmed.is_empty() {
419            if !prev_blank {
420                result.push("");
421                prev_blank = true;
422            }
423            continue;
424        }
425        prev_blank = false;
426        result.push(line);
427    }
428    let trailing = input.ends_with('\n');
429    let mut out = result.join("\n");
430    if trailing { out.push('\n'); }
431    if out.len() < input.len() { out } else { input.to_string() }
432}

434/// Word abbreviation table (subset of the Rust engine's 100+ entries).
435fn abbreviate_words(text: &str) -> String {
436    let abbrevs: &[(&str, &str)] = &[
437        ("implementation", "impl"),
438        ("configuration", "config"),
439        ("authentication", "auth"),
440        ("authorization", "authz"),
441        ("application", "app"),
442        ("environment", "env"),
443        ("development", "dev"),
444        ("production", "prod"),
445        ("repository", "repo"),
446        ("dependency", "dep"),
447        ("dependencies", "deps"),
448        ("documentation", "docs"),
449        ("information", "info"),
450        ("directory", "dir"),
451        ("parameter", "param"),
452        ("parameters", "params"),
453        ("function", "fn"),
454        ("reference", "ref"),
455        ("specification", "spec"),
456        ("administrator", "admin"),
457        ("database", "db"),
458        ("infrastructure", "infra"),
459        ("kubernetes", "k8s"),
460        ("namespace", "ns"),
461        ("management", "mgmt"),
462        ("notification", "notif"),
463        ("permission", "perm"),
464        ("permissions", "perms"),
465    ];
466    let mut result = text.to_string();
467    for &(long, short) in abbrevs {
468        // Simple whole-word replacement (case-insensitive)
469        result = replace_word_ci(&result, long, short);
470    }
471    result
472}

474fn replace_word_ci(text: &str, word: &str, replacement: &str) -> String {
475    let lower = text.to_lowercase();
476    let word_lower = word.to_lowercase();
477    let mut result = String::with_capacity(text.len());
478    let mut last = 0;
479    let bytes = text.as_bytes();
480    for (start, _) in lower.match_indices(&word_lower) {
481        let end = start + word.len();
482        let before_ok = start == 0 || !bytes[start - 1].is_ascii_alphanumeric();
483        let after_ok = end >= text.len() || !bytes[end].is_ascii_alphanumeric();
484        if before_ok && after_ok {
485            result.push_str(&text[last..start]);
486            result.push_str(replacement);
487            last = end;
488        }
489    }
490    result.push_str(&text[last..]);
491    result
492}

498fn content_hash(text: &str) -> u64 {
499    use std::hash::{Hash, Hasher};
500    let mut hasher = std::collections::hash_map::DefaultHasher::new();
501    text.hash(&mut hasher);
502    hasher.finish()
503}

515// ---------------------------------------------------------------------------
516// Compression preset
517// ---------------------------------------------------------------------------
518
519#[derive(Debug, Clone, Copy, PartialEq, Eq)]
520enum Preset {
521    /// TOON only, no stripping.
522    Minimal,
523    /// Strip nulls + condense + TOON + word abbreviation.
524    Default,
525    /// All of Default + collapse arrays + aggressive text compression.
526    Aggressive,
527}
528
529impl Preset {
530    fn from_str(s: &str) -> Self {
531        match s.to_lowercase().as_str() {
532            "minimal" => Preset::Minimal,
533            "aggressive" => Preset::Aggressive,
534            _ => Preset::Default,
535        }
536    }
537}

539// ---------------------------------------------------------------------------
540// WasmEngine — full-featured subset for browser use
541// ---------------------------------------------------------------------------
542
543struct WasmEngine {
544    session: SessionState,
545    preset: Preset,
546    /// In-memory dedup cache: hash → compressed output.
547    dedup_cache: HashMap<u64, String>,
548}
549
550impl WasmEngine {
551    fn new(preset: Preset) -> Self {
552        Self {
553            session: SessionState::default(),
554            preset,
555            dedup_cache: HashMap::new(),
556        }
557    }
558
559    fn compress(&mut self, input: &str) -> String {
560        // Dedup check: if we've seen this exact content before, return ref
561        let hash = content_hash(input);
562        if let Some(cached) = self.dedup_cache.get(&hash) {
563            return cached.clone();
564        }
565
566        let result = self.compress_inner(input);
567
568        // Cache the result for future dedup
569        self.dedup_cache.insert(hash, result.clone());
570        result
571    }
572
573    fn compress_inner(&self, input: &str) -> String {
574        let content_type = classify(input);
575
576        match content_type {
577            ContentType::Json => self.compress_json(input),
578            ContentType::Code => compress_code(input),
579            ContentType::Log => compress_log(input),
580            ContentType::Prose => compress_text(input),
581        }
582    }
583
584    fn compress_json(&self, input: &str) -> String {
585        let parsed = match serde_json::from_str::<serde_json::Value>(input.trim()) {
586            Ok(v) => v,
587            Err(_) => return input.to_string(),
588        };
589
590        let mut value = parsed;
591
592        // Strip nulls (unless Minimal preset)
593        if self.preset != Preset::Minimal {
594            strip_nulls(&mut value);
595        }
596
597        // Collapse arrays (Aggressive only, or Default with large arrays)
598        match self.preset {
599            Preset::Aggressive => collapse_arrays(&mut value, 5),
600            Preset::Default => collapse_arrays(&mut value, 10),
601            Preset::Minimal => {}
602        }
603
604        // TOON encode
605        let mut buf = String::with_capacity(input.len());
606        buf.push_str(TOON_PREFIX);
607        encode_value(&value, &mut buf);
608
609        if buf.len() < input.len() { buf } else { input.to_string() }
610    }
611
612    fn estimate_tokens(&self, input: &str) -> u32 {
613        (input.len() as f64 / 4.0).ceil() as u32
614    }
615
616    fn export_ctx(&self) -> Result<String, String> {
617        serde_json::to_string(&self.session).map_err(|e| e.to_string())
618    }
619
620    fn import_ctx(&mut self, ctx: &str) -> Result<(), String> {
621        self.session = serde_json::from_str(ctx).map_err(|e| e.to_string())?;
622        Ok(())
623    }
624}

630/// Browser-facing WASM wrapper for the sqz compression engine.
631///
632/// Provides content-aware compression (JSON, code, logs, prose), TOON
633/// encoding, word abbreviation, in-memory dedup caching, and session
634/// export/import. No filesystem or network access — everything runs in memory.
635#[wasm_bindgen]
636pub struct SqzWasm {
637    engine: WasmEngine,
638}
639
640#[wasm_bindgen]
641impl SqzWasm {
642    /// Create a new SqzWasm instance.
643    /// `preset_json` selects the compression preset:
644    /// - `"minimal"` — TOON encoding only
645    /// - `"default"` — strip nulls + condense + TOON + word abbreviation
646    /// - `"aggressive"` — all of default + collapse arrays at 5 items
647    #[wasm_bindgen(constructor)]
648    pub fn new(preset_json: &str) -> Result<SqzWasm, JsValue> {
649        let preset = Preset::from_str(preset_json);
650        Ok(SqzWasm {
651            engine: WasmEngine::new(preset),
652        })
653    }
654
655    /// Compress `input`. Routes to JSON/code/log/prose compressor based on
656    /// content classification. Returns cached result on duplicate input.
657    pub fn compress(&mut self, input: &str) -> Result<JsValue, JsValue> {
658        let compressed = self.engine.compress(input);
659        Ok(JsValue::from_str(&compressed))
660    }
661
662    /// Estimate the token count for `input` using the GPT-style approximation
663    /// (chars / 4, rounded up).
664    pub fn estimate_tokens(&self, input: &str) -> u32 {
665        self.engine.estimate_tokens(input)
666    }
667
668    /// Serialize the current session state to a JSON string.
669    pub fn export_ctx(&self) -> Result<String, JsValue> {
670        self.engine.export_ctx().map_err(|e| JsValue::from_str(&e))
671    }
672
673    /// Deserialize session state from a JSON string produced by `export_ctx`.
674    pub fn import_ctx(&mut self, ctx: &str) -> Result<(), JsValue> {
675        self.engine
676            .import_ctx(ctx)
677            .map_err(|e| JsValue::from_str(&e))
678    }
679}
