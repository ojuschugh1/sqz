3/// Token counter using BPE tokenizer models.
4///
5/// Supports:
6/// - OpenAI GPT-4/4o: exact via `cl100k_base`
7/// - OpenAI o1/o3: exact via `o200k_base`
8/// - Anthropic Claude: approximate via `cl100k_base` (Claude uses a proprietary tokenizer;
9///   cl100k_base is a close approximation, typically within 5% of actual count)
10/// - Google Gemini: approximate via `cl100k_base` (SentencePiece-based; cl100k_base provides
11///   a reasonable cross-model estimate)
12/// - Local/unknown models: fast heuristic fallback (`chars / 4`)
13///
14/// Uses `tiktoken-rs` singletons so the BPE data is loaded once and reused.
15pub struct TokenCounter;
16
17impl TokenCounter {
18    /// Create a new `TokenCounter`.
19    ///
20    /// The underlying tokenizers are lazily-initialized singletons from
21    /// `tiktoken-rs`, so construction is essentially free.
22    pub fn new() -> Self {
23        Self
24    }
25
26    /// Count tokens for `text` using the tokenizer appropriate for `model`.
27    ///
28    /// - `OpenAiGpt` → `o200k_base` (exact for GPT-4o / o1 / o3)
29    /// - `AnthropicClaude` → `cl100k_base` (close approximation, ~5% variance)
30    /// - `GoogleGemini` → `cl100k_base` (cross-model approximation)
31    /// - `Local(_)` → `chars / 4` heuristic fallback
32    pub fn count(&self, text: &str, model: &ModelFamily) -> u32 {
33        match model {
34            ModelFamily::OpenAiGpt => self.count_o200k(text),
35            ModelFamily::AnthropicClaude | ModelFamily::GoogleGemini => self.count_cl100k(text),
36            ModelFamily::Local(_) => Self::count_fast(text),
37        }
38    }
39
40    /// Fast character-based approximation: `ceil(chars / 4)`.
41    ///
42    /// Used as the fallback for unknown or local models.
43    pub fn count_fast(text: &str) -> u32 {
44        ((text.len() as f64) / 4.0).ceil() as u32
45    }
46
47    /// Count tokens using the `cl100k_base` encoding (GPT-4 / Claude compatible).
48    fn count_cl100k(&self, text: &str) -> u32 {
49        let bpe = tiktoken_rs::cl100k_base_singleton();
50        let lock = bpe.lock();
51        lock.encode_with_special_tokens(text).len() as u32
52    }
53
54    /// Count tokens using the `o200k_base` encoding (GPT-4o / o1 / o3).
55    fn count_o200k(&self, text: &str) -> u32 {
56        let bpe = tiktoken_rs::o200k_base_singleton();
57        let lock = bpe.lock();
58        lock.encode_with_special_tokens(text).len() as u32
59    }
60}
61
62impl Default for TokenCounter {
63    fn default() -> Self {
64        Self::new()
65    }
66}

134    use proptest::prelude::*;
135
136    /// Strategy that produces arbitrary non-empty strings.
137    fn arb_nonempty_text() -> impl Strategy<Value = String> {
138        // Mix of ASCII, unicode, code-like, and whitespace-heavy strings
139        prop_oneof![
140            // General unicode strings (1..256 chars)
141            "\\PC{1,256}",
142            // ASCII-only strings
143            "[a-zA-Z0-9 .,;:!?()\\[\\]{}<>@#$%^&*_+=~`|/\\\\-]{1,256}",
144            // Code-like strings
145            "(fn|def|class|import|const|let|var|pub|async|await|return)[ a-zA-Z0-9_(){}:;,\n]{1,200}",
146        ]
147        .prop_filter("must be non-empty", |s| !s.is_empty())
148    }
149
150    /// Strategy for known model families (not Local, which uses fallback).
151    fn arb_known_model() -> impl Strategy<Value = ModelFamily> {
152        prop_oneof![
153            Just(ModelFamily::OpenAiGpt),
154            Just(ModelFamily::AnthropicClaude),
155            Just(ModelFamily::GoogleGemini),
156        ]
157    }

159    proptest! {
160        /// **Validates: Requirements 32.1, 32.2**
161        ///
162        /// Property 37a: For any non-empty text and known model family,
163        /// the exact token count is always > 0.
164        #[test]
165        fn prop37a_exact_count_positive_for_nonempty(
166            text in arb_nonempty_text(),
167            model in arb_known_model(),
168        ) {
169            let tc = TokenCounter::new();
170            let count = tc.count(&text, &model);
171            prop_assert!(
172                count > 0,
173                "exact token count must be > 0 for non-empty string (len={}), got {}",
174                text.len(),
175                count
176            );
177        }
178
179        /// **Validates: Requirements 32.1, 32.2**
180        ///
181        /// Property 37b: For any non-empty text and known model family,
182        /// the exact token count is always ≤ the character count, since
183        /// BPE tokens span one or more characters.
184        #[test]
185        fn prop37b_exact_count_leq_char_count(
186            text in arb_nonempty_text(),
187            model in arb_known_model(),
188        ) {
189            let tc = TokenCounter::new();
190            let count = tc.count(&text, &model);
191            let char_count = text.len() as u32;
192            prop_assert!(
193                count <= char_count,
194                "exact token count ({}) must be <= byte length ({}) for text",
195                count,
196                char_count
197            );
198        }
199
200        /// **Validates: Requirements 32.1, 32.2**
201        ///
202        /// Property 37c: The fast fallback (chars/4) is within a
203        /// reasonable factor (0.1x–10x) of the exact count for
204        /// non-trivial text (>= 8 chars).
205        #[test]
206        fn prop37c_fast_fallback_within_reasonable_factor(
207            text in "[a-zA-Z0-9 .,;:!?]{8,256}",
208            model in arb_known_model(),
209        ) {
210            let tc = TokenCounter::new();
211            let exact = tc.count(&text, &model) as f64;
212            let fast = TokenCounter::count_fast(&text) as f64;
213            // Both should be positive
214            prop_assert!(exact > 0.0);
215            prop_assert!(fast > 0.0);
216            let ratio = exact / fast;
217            prop_assert!(
218                ratio >= 0.1 && ratio <= 10.0,
219                "exact/fast ratio ({:.2}) out of reasonable range [0.1, 10.0]; exact={}, fast={}",
220                ratio,
221                exact,
222                fast
223            );
224        }
225    }

227    #[test]
228    fn performance_10k_tokens_under_1ms() {
229        let tc = TokenCounter::new();
230        // ~40K chars ≈ 10K tokens for typical English text
231        let text = "Hello world. This is a test sentence for benchmarking. ".repeat(800);
232
233        // Warm up the singleton
234        let _ = tc.count(&text, &ModelFamily::OpenAiGpt);
235
236        let start = std::time::Instant::now();
237        let _count = tc.count(&text, &ModelFamily::OpenAiGpt);
238        let elapsed = start.elapsed();
239
240        // In debug builds under test load, allow generous headroom.
241        // The <1ms target (Req 32.5) applies to optimized release builds.
242        assert!(
243            elapsed.as_millis() < 500,
244            "token counting took {}ms, expected <500ms (debug build)",
245            elapsed.as_millis()
246        );
247    }
