1//! Parse Tree Guided Code Compression — uses tree-sitter ASTs to identify
2//! low-entropy subtrees and collapse them to their signature line.
3//!
4//! For each node in the parse tree, we compute the Shannon entropy of its
5//! text content (character-level). Nodes with entropy below the median are
6//! considered "low-entropy" (boilerplate, repetitive code) and are collapsed
7//! to just their first line (the signature). High-entropy nodes are preserved
8//! in full.
9//!
10//! # Supported languages
11//! - Rust
12//! - Python
13//! - JavaScript
14
15use tree_sitter::{Language, Parser, Node};
16use crate::error::{Result, SqzError};
17
18/// Compress source code by collapsing low-entropy parse tree subtrees.
19///
20/// Parses `code` using tree-sitter for the given `language`, computes
21/// per-node character entropy, and collapses subtrees below the median
22/// entropy to their signature line (first line).
23///
24/// # Arguments
25/// * `code` — source code to compress
26/// * `language` — one of `"rust"`, `"python"`, `"javascript"`
27///
28/// # Returns
29/// Compressed code string with low-entropy subtrees collapsed.
30pub fn compress_code(code: &str, language: &str) -> Result<String> {
31    let ts_lang = get_language(language)?;
32    let mut parser = Parser::new();
33    parser
34        .set_language(&ts_lang)
35        .map_err(|e| SqzError::Other(format!("tree-sitter language error: {e}")))?;
36
37    let tree = parser
38        .parse(code, None)
39        .ok_or_else(|| SqzError::Other("tree-sitter parse failed".into()))?;
40
41    let root = tree.root_node();
42
43    // Collect top-level nodes with their entropy
44    let mut node_entropies: Vec<(usize, usize, f64)> = Vec::new(); // (start_byte, end_byte, entropy)
45    let mut cursor = root.walk();
46    for child in root.children(&mut cursor) {
47        let text = &code[child.byte_range()];
48        let entropy = char_entropy(text);
49        node_entropies.push((child.start_byte(), child.end_byte(), entropy));
50    }
51
52    if node_entropies.is_empty() {
53        return Ok(code.to_string());
54    }
55
56    // Compute median entropy
57    let median = compute_median_entropy(&node_entropies);
58
59    // Build output: preserve high-entropy nodes, collapse low-entropy ones
60    let mut result = String::new();
61    let mut last_end = 0;
62
63    for (start, end, entropy) in &node_entropies {
64        // Preserve whitespace/comments between nodes
65        if *start > last_end {
66            result.push_str(&code[last_end..*start]);
67        }
68
69        let node_text = &code[*start..*end];
70        if *entropy < median {
71            // Collapse to signature line
72            let sig = signature_line(node_text);
73            result.push_str(&sig);
74            result.push_str(" /* … */");
75        } else {
76            result.push_str(node_text);
77        }
78
79        last_end = *end;
80    }
81
82    // Preserve trailing content
83    if last_end < code.len() {
84        result.push_str(&code[last_end..]);
85    }
86
87    Ok(result)
88}

90/// Compute Shannon entropy (bits per character) of a string.
91///
92/// Uses character frequency distribution. Returns 0.0 for empty strings.
93pub fn char_entropy(text: &str) -> f64 {
94    if text.is_empty() {
95        return 0.0;
96    }
97
98    let mut freq = [0u32; 256];
99    let mut total = 0u32;
100
101    for byte in text.bytes() {
102        freq[byte as usize] += 1;
103        total += 1;
104    }
105
106    if total == 0 {
107        return 0.0;
108    }
109
110    let total_f = total as f64;
111    let mut entropy = 0.0;
112    for &count in &freq {
113        if count > 0 {
114            let p = count as f64 / total_f;
115            entropy -= p * p.log2();
116        }
117    }
118
119    entropy
120}

122/// Extract the first line of a node's text as its "signature".
123fn signature_line(text: &str) -> String {
124    text.lines()
125        .next()
126        .unwrap_or("")
127        .to_string()
128}

130/// Compute the median entropy from a list of (start, end, entropy) tuples.
131fn compute_median_entropy(entries: &[(usize, usize, f64)]) -> f64 {
132    let mut entropies: Vec<f64> = entries.iter().map(|(_, _, e)| *e).collect();
133    entropies.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
134    let mid = entropies.len() / 2;
135    if entropies.len() % 2 == 0 && entropies.len() >= 2 {
136        (entropies[mid - 1] + entropies[mid]) / 2.0
137    } else {
138        entropies[mid]
139    }
140}

142/// Resolve a language name to a tree-sitter `Language`.
143fn get_language(language: &str) -> Result<Language> {
144    match language {
145        "rust" => Ok(tree_sitter_rust::language()),
146        "python" => Ok(tree_sitter_python::language()),
147        "javascript" | "js" => Ok(tree_sitter_javascript::language()),
148        other => Err(SqzError::UnsupportedLanguage(other.to_string())),
149    }
150}

152/// Recursively collect entropy for all named children of a node.
153/// Returns a flat list of (node_kind, start_byte, end_byte, entropy).
154#[allow(dead_code)]
155fn collect_node_entropies<'a>(
156    node: &Node<'a>,
157    source: &str,
158) -> Vec<(String, usize, usize, f64)> {
159    let mut results = Vec::new();
160    let mut cursor = node.walk();
161    for child in node.children(&mut cursor) {
162        if child.is_named() {
163            let text = &source[child.byte_range()];
164            let entropy = char_entropy(text);
165            results.push((
166                child.kind().to_string(),
167                child.start_byte(),
168                child.end_byte(),
169                entropy,
170            ));
171        }
172    }
173    results
174}
