1/// Dictionary-based JSON Compression.
2///
3/// Exploits cross-document redundancy by maintaining a dictionary of common
4/// JSON field names, value patterns, and structural elements. Each new JSON
5/// document is compressed relative to this shared dictionary, achieving
6/// better ratios than compressing in isolation.
7///
8/// This is a pure-Rust implementation that avoids adding the zstd crate
9/// (which would add ~200KB to the binary). Instead, it uses a dictionary
10/// substitution approach optimized for JSON payloads.
11
12use std::collections::HashMap;
13
14use crate::error::Result;
15
16/// A dictionary entry mapping a common pattern to a short code.
17#[derive(Debug, Clone)]
18struct DictEntry {
19    /// The original pattern (e.g., a field name or common value).
20    pattern: String,
21    /// The short replacement code.
22    code: String,
23    /// How many times this pattern has been seen.
24    #[allow(dead_code)]
25    frequency: u32,
26}
27
28/// Configuration for dictionary compression.
29#[derive(Debug, Clone)]
30pub struct DictConfig {
31    /// Maximum number of dictionary entries.
32    /// Default: 256
33    pub max_entries: usize,
34    /// Minimum pattern length (chars) to consider for dictionary.
35    /// Default: 4
36    pub min_pattern_length: usize,
37    /// Minimum frequency before a pattern is added to the dictionary.
38    /// Default: 2
39    pub min_frequency: u32,
40}
41
42impl Default for DictConfig {
43    fn default() -> Self {
44        Self {
45            max_entries: 256,
46            min_pattern_length: 4,
47            min_frequency: 2,
48        }
49    }
50}
51
52/// Result of dictionary compression.
53#[derive(Debug, Clone)]
54pub struct DictCompressResult {
55    /// The compressed text.
56    pub data: String,
57    /// Number of substitutions made.
58    pub substitutions: usize,
59    /// Bytes saved.
60    pub bytes_saved: usize,
61    /// Whether the dictionary header was included.
62    pub has_dict_header: bool,
63}
64
65/// Dictionary-based JSON compressor.
66///
67/// Maintains a session-level dictionary of common JSON patterns and
68/// substitutes them with short codes during compression.
69pub struct DictCompressor {
70    config: DictConfig,
71    /// Pattern → DictEntry mapping.
72    entries: HashMap<String, DictEntry>,
73    /// Frequency counter for candidate patterns.
74    candidates: HashMap<String, u32>,
75    /// Next code index for new entries.
76    next_code: u16,
77    /// Built-in common JSON field names.
78    builtin_entries: Vec<DictEntry>,
79}
80
81impl DictCompressor {
82    pub fn new() -> Self {
83        Self::with_config(DictConfig::default())
84    }
85
86    pub fn with_config(config: DictConfig) -> Self {
87        let builtin = build_builtin_dictionary();
88        let mut compressor = Self {
89            config,
90            entries: HashMap::new(),
91            candidates: HashMap::new(),
92            next_code: 0,
93            builtin_entries: builtin,
94        };
95        compressor.load_builtins();
96        compressor
97    }
98
99    /// Load built-in dictionary entries.
100    fn load_builtins(&mut self) {
101        for entry in &self.builtin_entries.clone() {
102            self.entries.insert(entry.pattern.clone(), entry.clone());
103        }
104        self.next_code = self.entries.len() as u16;
105    }
106
107    /// Observe a JSON string to learn common patterns.
108    pub fn observe(&mut self, json_str: &str) {
109        // Extract field names from JSON
110        let fields = extract_json_fields(json_str);
111        for field in fields {
112            if field.len() >= self.config.min_pattern_length {
113                let count = self.candidates.entry(field.clone()).or_insert(0);
114                *count += 1;
115
116                // Promote to dictionary if frequency threshold met
117                if *count >= self.config.min_frequency
118                    && !self.entries.contains_key(&field)
119                    && self.entries.len() < self.config.max_entries
120                {
121                    let code = format!("~{:X}", self.next_code);
122                    self.next_code += 1;
123                    self.entries.insert(
124                        field.clone(),
125                        DictEntry {
126                            pattern: field,
127                            code,
128                            frequency: *count,
129                        },
130                    );
131                }
132            }
133        }
134    }
135
136    /// Compress a JSON string using the dictionary.
137    ///
138    /// Substitutes known field names and patterns with short codes.
139    /// Prepends a dictionary header if substitutions were made.
140    pub fn compress(&self, json_str: &str) -> Result<DictCompressResult> {
141        if self.entries.is_empty() {
142            return Ok(DictCompressResult {
143                data: json_str.to_string(),
144                substitutions: 0,
145                bytes_saved: 0,
146                has_dict_header: false,
147            });
148        }
149
150        let mut result = json_str.to_string();
151        let mut substitutions = 0usize;
152        let mut bytes_saved = 0usize;
153        let mut used_entries: Vec<&DictEntry> = Vec::new();
154
155        // Sort entries by pattern length (longest first) to avoid partial matches
156        let mut sorted_entries: Vec<&DictEntry> = self.entries.values().collect();
157        sorted_entries.sort_by(|a, b| b.pattern.len().cmp(&a.pattern.len()));
158
159        for entry in &sorted_entries {
160            // Only substitute in JSON key positions: "field_name":
161            let search = format!("\"{}\":", entry.pattern);
162            let replace = format!("\"{}\":", entry.code);
163
164            if result.contains(&search) {
165                let count = result.matches(&search).count();
166                result = result.replace(&search, &replace);
167                substitutions += count;
168                bytes_saved += count * (search.len() - replace.len());
169                used_entries.push(entry);
170            }
171        }
172
173        // Only add header if we actually made substitutions
174        let has_dict_header = !used_entries.is_empty();
175        if has_dict_header {
176            let header = format_dict_header(&used_entries);
177            result = format!("{header}\n{result}");
178        }
179
180        Ok(DictCompressResult {
181            data: result,
182            substitutions,
183            bytes_saved,
184            has_dict_header,
185        })
186    }
187
188    /// Get the current dictionary size.
189    pub fn dict_size(&self) -> usize {
190        self.entries.len()
191    }
192
193    /// Reset the dictionary (e.g., on session reset).
194    pub fn reset(&mut self) {
195        self.entries.clear();
196        self.candidates.clear();
197        self.next_code = 0;
198        self.load_builtins();
199    }
200}
201
202impl Default for DictCompressor {
203    fn default() -> Self {
204        Self::new()
205    }
206}

210/// Build the default dictionary of common JSON field names.
211fn build_builtin_dictionary() -> Vec<DictEntry> {
212    let common_fields = [
213        // API response fields
214        ("status_code", "~s"),
215        ("message", "~m"),
216        ("timestamp", "~ts"),
217        ("created_at", "~ca"),
218        ("updated_at", "~ua"),
219        ("deleted_at", "~da"),
220        ("description", "~dc"),
221        ("metadata", "~md"),
222        ("properties", "~pp"),
223        ("attributes", "~at"),
224        ("parameters", "~pm"),
225        ("configuration", "~cf"),
226        ("environment", "~ev"),
227        ("dependencies", "~dp"),
228        ("permissions", "~pr"),
229        ("resources", "~rs"),
230        ("namespace", "~ns"),
231        ("annotations", "~an"),
232        // Error fields
233        ("error_code", "~ec"),
234        ("error_message", "~em"),
235        ("stack_trace", "~st"),
236        ("exception", "~ex"),
237        // Pagination
238        ("page_size", "~ps"),
239        ("next_token", "~nt"),
240        ("total_count", "~tc"),
241        // AWS/Cloud common
242        ("resource_type", "~rt"),
243        ("account_id", "~ai"),
244        ("region", "~rg"),
245    ];
246
247    common_fields
248        .iter()
249        .enumerate()
250        .map(|(_, &(pattern, code))| DictEntry {
251            pattern: pattern.to_string(),
252            code: code.to_string(),
253            frequency: 100, // built-in entries have high base frequency
254        })
255        .collect()
256}

258/// Extract field names from a JSON string (simple regex-free parser).
259fn extract_json_fields(json_str: &str) -> Vec<String> {
260    let mut fields = Vec::new();
261    let bytes = json_str.as_bytes();
262    let len = bytes.len();
263    let mut i = 0;
264
265    while i < len {
266        // Look for "field_name":
267        if bytes[i] == b'"' {
268            let start = i + 1;
269            i += 1;
270            // Find closing quote
271            while i < len && bytes[i] != b'"' {
272                if bytes[i] == b'\\' {
273                    i += 1; // skip escaped char
274                }
275                i += 1;
276            }
277            if i < len {
278                let end = i;
279                i += 1;
280                // Check if followed by ':'
281                // Skip whitespace
282                while i < len && (bytes[i] == b' ' || bytes[i] == b'\t' || bytes[i] == b'\n' || bytes[i] == b'\r') {
283                    i += 1;
284                }
285                if i < len && bytes[i] == b':' {
286                    if let Ok(field) = std::str::from_utf8(&bytes[start..end]) {
287                        fields.push(field.to_string());
288                    }
289                }
290            }
291        } else {
292            i += 1;
293        }
294    }
295
296    fields
297}

299/// Format the dictionary header for compressed output.
300fn format_dict_header(entries: &[&DictEntry]) -> String {
301    let mut lines = vec!["§dict§".to_string()];
302    for entry in entries {
303        lines.push(format!("{}={}", entry.code, entry.pattern));
304    }
305    lines.push("§/dict§".to_string());
306    lines.join("\n")
307}
