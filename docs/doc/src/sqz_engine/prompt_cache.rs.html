3/// A detected prompt cache boundary in a message sequence.
4#[derive(Debug, Clone, PartialEq)]
5pub struct CacheBoundary {
6    /// Byte offset in the concatenated content string where the boundary falls.
7    /// Content before this offset is cached; content at/after is not.
8    pub offset: usize,
9    /// The provider that owns this cache boundary.
10    pub provider: Provider,
11    /// The cache discount fraction (0.9 for Anthropic, 0.5 for OpenAI).
12    pub discount: f64,
13}
14
15/// A single message in a conversation, used for cache boundary detection.
16#[derive(Debug, Clone)]
17pub struct Message {
18    pub role: String,
19    pub content: String,
20    /// Anthropic-style cache control marker.
21    /// When `Some("ephemeral")`, this message has a cache_control marker.
22    pub cache_control: Option<String>,
23}
24
25/// Detects prompt cache boundaries for Anthropic and OpenAI API formats.
26///
27/// - **Anthropic**: uses `cache_control: {"type": "ephemeral"}` markers on
28///   individual messages. The boundary is placed after the last marked message.
29/// - **OpenAI**: automatically caches the first 1024 tokens of a prompt.
30///   Detected heuristically: if total content length > 4096 chars (≈1024 tokens),
31///   the boundary is at char position 4096.
32/// - **Google**: no cache boundary concept; always returns `None`.
33pub struct PromptCacheDetector;
34
35impl PromptCacheDetector {
36    /// Detect a cache boundary in the given message sequence for the specified
37    /// provider. Returns `None` if no boundary is detected.
38    pub fn detect_boundary(
39        &self,
40        messages: &[Message],
41        provider: Provider,
42    ) -> Option<CacheBoundary> {
43        match provider {
44            Provider::Anthropic => self.detect_anthropic(messages),
45            Provider::OpenAI => self.detect_openai(messages),
46            Provider::Google => None,
47        }
48    }
49
50    /// Split `content` at the given boundary offset.
51    ///
52    /// Returns `(before, after)` where `before` is byte-identical to the
53    /// original content up to `boundary.offset`, and `after` is the remainder.
54    ///
55    /// If `boundary.offset >= content.len()`, `after` will be empty.
56    pub fn split_at_boundary(&self, content: &str, boundary: &CacheBoundary) -> (String, String) {
57        let offset = boundary.offset.min(content.len());
58        let before = content[..offset].to_owned();
59        let after = content[offset..].to_owned();
60        (before, after)
61    }
62
63    // -----------------------------------------------------------------------
64    // Private helpers
65    // -----------------------------------------------------------------------
66
67    fn detect_anthropic(&self, messages: &[Message]) -> Option<CacheBoundary> {
68        // Find the last message that carries a cache_control marker.
69        // The boundary offset is the byte position after that message's content.
70        let mut offset: usize = 0;
71        let mut boundary_offset: Option<usize> = None;
72
73        for msg in messages {
74            offset += msg.content.len();
75            if msg.cache_control.as_deref() == Some("ephemeral") {
76                boundary_offset = Some(offset);
77            }
78        }
79
80        boundary_offset.map(|off| CacheBoundary {
81            offset: off,
82            provider: Provider::Anthropic,
83            discount: 0.9,
84        })
85    }
86
87    fn detect_openai(&self, messages: &[Message]) -> Option<CacheBoundary> {
88        // OpenAI automatically caches the first 1024 tokens.
89        // Heuristic: if total content length > 4096 chars, the boundary is at
90        // char position 4096 (≈1024 tokens at ~4 chars/token).
91        const OPENAI_CACHE_CHAR_THRESHOLD: usize = 4096;
92
93        let total_len: usize = messages.iter().map(|m| m.content.len()).sum();
94        if total_len > OPENAI_CACHE_CHAR_THRESHOLD {
95            Some(CacheBoundary {
96                offset: OPENAI_CACHE_CHAR_THRESHOLD,
97                provider: Provider::OpenAI,
98                discount: 0.5,
99            })
100        } else {
101            None
102        }
103    }
104}

293    /// Strategy: generate a list of messages where at least one has a
294    /// cache_control marker (Anthropic format).
295    fn anthropic_messages_with_boundary() -> impl Strategy<Value = Vec<Message>> {
296        // Generate 1-8 messages; at least one will be marked as cached.
297        (1usize..=8usize).prop_flat_map(|n| {
298            // For each message, generate content and whether it's cached.
299            let msg_strategy = (
300                prop_oneof![Just("user"), Just("assistant"), Just("system")],
301                "[a-z]{1,50}",
302                any::<bool>(),
303            );
304            prop::collection::vec(msg_strategy, n).prop_filter(
305                "at least one cached message",
306                |msgs| msgs.iter().any(|(_, _, cached)| *cached),
307            )
308        })
309        .prop_map(|msgs| {
310            msgs.into_iter()
311                .map(|(role, content, cached)| Message {
312                    role: role.to_owned(),
313                    content,
314                    cache_control: if cached {
315                        Some("ephemeral".to_owned())
316                    } else {
317                        None
318                    },
319                })
320                .collect()
321        })
322    }

324    proptest! {
325        /// **Validates: Requirements 4.1, 4.2**
326        ///
327        /// Property 3: Prompt cache boundary preservation.
328        ///
329        /// For any message sequence containing a prompt cache boundary marker
330        /// (Anthropic or OpenAI format), the Prompt_Cache_Detector SHALL
331        /// identify the boundary, and content preceding the boundary SHALL be
332        /// byte-identical to the original.
333        #[test]
334        fn prop_prompt_cache_boundary_preservation(
335            messages in anthropic_messages_with_boundary(),
336        ) {
337            let detector = PromptCacheDetector;
338
339            // 1. Boundary must be detected (at least one cached message exists).
340            let boundary = detector.detect_boundary(&messages, Provider::Anthropic);
341            prop_assert!(
342                boundary.is_some(),
343                "expected boundary to be detected for messages with cache_control markers"
344            );
345            let boundary = boundary.unwrap();
346
347            // 2. Discount must be 0.9 for Anthropic.
348            prop_assert_eq!(boundary.discount, 0.9);
349
350            // 3. Content before the boundary must be byte-identical to the original.
351            //    Concatenate all message contents to form the "full content".
352            let full_content: String = messages.iter().map(|m| m.content.as_str()).collect();
353            let (before, _after) = detector.split_at_boundary(&full_content, &boundary);
354
355            let expected_prefix = &full_content[..boundary.offset.min(full_content.len())];
356            prop_assert_eq!(
357                before.as_bytes(),
358                expected_prefix.as_bytes(),
359                "content before boundary must be byte-identical to original prefix"
360            );
361
362            // 4. The boundary offset must be at or after the last cached message's
363            //    content end, and must not exceed the total content length.
364            prop_assert!(
365                boundary.offset <= full_content.len(),
366                "boundary offset {} exceeds content length {}",
367                boundary.offset,
368                full_content.len()
369            );
370
371            // 5. Verify the boundary is placed after the last cached message.
372            let mut cumulative = 0usize;
373            let mut last_cached_end = 0usize;
374            for msg in &messages {
375                cumulative += msg.content.len();
376                if msg.cache_control.as_deref() == Some("ephemeral") {
377                    last_cached_end = cumulative;
378                }
379            }
380            prop_assert_eq!(
381                boundary.offset,
382                last_cached_end,
383                "boundary offset should be after the last cached message"
384            );
385        }
386    }
