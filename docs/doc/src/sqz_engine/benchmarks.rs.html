1/// sqz compression benchmark suite.
2///
3/// Measures faithfulness, critical-info retention, and token reduction
4/// across representative content types. Used as CI regression gates.
5///
6/// Run with: `cargo test -p sqz-engine benchmarks`
7
8#[cfg(test)]
9mod tests {
10    use crate::engine::SqzEngine;
11    use crate::verifier::Verifier;
12
13    // ── Helpers ──────────────────────────────────────────────────────────────
14
15    struct BenchResult {
16        content_type: &'static str,
17        original_tokens: u32,
18        compressed_tokens: u32,
19        reduction_pct: f64,
20        verify_confidence: f64,
21        critical_info_retained: bool,
22    }
23
24    fn estimate_tokens(s: &str) -> u32 {
25        ((s.len() as f64) / 4.0).ceil() as u32
26    }
27
28    fn run_bench(content_type: &'static str, input: &str, critical_marker: &str) -> BenchResult {
29        // Use a temp DB per test to avoid SQLite locking in parallel CI
30        let dir = tempfile::tempdir().expect("tempdir");
31        let store_path = dir.path().join("bench.db");
32        let preset = crate::preset::Preset::default();
33        let engine = SqzEngine::with_preset_and_store(preset, &store_path).expect("engine init");
34        let result = engine.compress(input).expect("compress");
35
36        let original_tokens = estimate_tokens(input);
37        let compressed_tokens = estimate_tokens(&result.data);
38        let reduction_pct = (1.0 - compressed_tokens as f64 / original_tokens.max(1) as f64) * 100.0;
39
40        let verify = result.verify.unwrap_or_else(|| Verifier::verify(input, &result.data));
41        let critical_info_retained = result.data.contains(critical_marker);
42
43        BenchResult {
44            content_type,
45            original_tokens,
46            compressed_tokens,
47            reduction_pct,
48            verify_confidence: verify.confidence,
49            critical_info_retained,
50        }
51    }
52
53    fn assert_gates(b: &BenchResult, min_reduction: f64, min_confidence: f64) {
54        assert!(
55            b.critical_info_retained,
56            "[{}] critical info marker missing from output",
57            b.content_type
58        );
59        assert!(
60            b.verify_confidence >= min_confidence,
61            "[{}] verify confidence {:.2} < threshold {:.2}",
62            b.content_type, b.verify_confidence, min_confidence
63        );
64        // Only assert reduction if input is large enough to compress meaningfully
65        if b.original_tokens > 50 {
66            assert!(
67                b.reduction_pct >= min_reduction,
68                "[{}] reduction {:.1}% < threshold {:.1}%",
69                b.content_type, b.reduction_pct, min_reduction
70            );
71        }
72        println!(
73            "[bench] {} | {}→{} tokens | {:.1}% reduction | confidence {:.2}",
74            b.content_type, b.original_tokens, b.compressed_tokens,
75            b.reduction_pct, b.verify_confidence
76        );
77    }

81    /// JSON API response — should compress via TOON + strip_nulls.
82    /// Gate: ≥5% reduction, confidence ≥0.8, key "id" retained.
83    #[test]
84    fn bench_json_api_response() {
85        let input = r#"{"id":42,"name":"Alice","email":"alice@example.com","role":"admin","created_at":"2024-01-01T00:00:00Z","updated_at":"2024-03-15T10:30:00Z","metadata":{"plan":"pro","seats":10,"billing_cycle":"monthly","internal_id":null,"debug_info":null,"trace_id":null}}"#;
86        let b = run_bench("json_api", input, "42");
87        assert_gates(&b, 5.0, 0.8);
88    }

90    /// Repeated log output — should compress via condense stage.

94        let repeated = "2024-01-01 10:00:00 [INFO] Connected to database\n".repeat(10);

102    /// Git diff — should fold unchanged context lines.
103    /// Gate: ≥10% reduction, confidence ≥0.8, hunk header retained.

105    fn bench_git_diff() {
106        let input = concat!(
107            "diff --git a/src/main.rs b/src/main.rs\n",
108            "--- a/src/main.rs\n",
109            "+++ b/src/main.rs\n",
110            "@@ -1,15 +1,15 @@\n",
111            " line1\n line2\n line3\n line4\n line5\n line6\n line7\n line8\n",
112            "-old_function_name\n",
113            "+new_function_name\n",
114            " line9\n line10\n line11\n line12\n line13\n line14\n line15\n",
115        );
116        let b = run_bench("git_diff", input, "@@ -1,15");
117        assert_gates(&b, 5.0, 0.8);
118    }

120    /// Plain prose — the Rust CLI engine does minimal prose compression
121    /// (it's optimized for JSON/logs). Verify no regression and critical info retained.
122    #[test]
123    fn bench_prose_documentation() {
124        let input = "In order to understand the system, it is important to note that the architecture consists of multiple components. Due to the fact that each component has the ability to operate independently, the system as a whole is resilient. In the event that one component fails, the other components continue to function. The vast majority of operations are handled by the core engine, which processes requests in real time. As a result of this design, the system achieves high availability and low latency.";
125        let b = run_bench("prose_docs", input, "architecture");
126        // Rust engine doesn't aggressively compress prose — just verify no data loss
127        assert!(b.critical_info_retained, "critical info must be retained");
128        assert!(
129            b.verify_confidence >= 0.7,
130            "[prose_docs] confidence {:.2} too low", b.verify_confidence
131        );
132        assert!(
133            b.compressed_tokens <= b.original_tokens,
134            "[prose_docs] output must not be larger than input"
135        );
136        println!(
137            "[bench] {} | {}→{} tokens | {:.1}% reduction | confidence {:.2}",
138            b.content_type, b.original_tokens, b.compressed_tokens,
139            b.reduction_pct, b.verify_confidence
140        );
141    }

143    /// Stack trace — should route to safe mode, preserve all error lines.
144    /// Gate: confidence ≥0.9 (safe mode), "panicked" retained.
145    #[test]
146    fn bench_stack_trace_safe_mode() {
147        let input = "thread 'main' panicked at 'index out of bounds: the len is 3 but the index is 5', src/main.rs:42\nnote: run with `RUST_BACKTRACE=1` environment variable to display a backtrace\nstack backtrace:\n   0: rust_begin_unwind\n   1: core::panicking::panic_fmt\n   2: core::slice::index_failed\n   3: myapp::process_items\n   4: myapp::main\n";
148        let b = run_bench("stack_trace", input, "panicked");
149        // Safe mode: high confidence, critical info preserved
150        assert!(b.critical_info_retained, "panicked marker must be retained");
151        assert!(
152            b.verify_confidence >= 0.7,
153            "stack trace confidence {:.2} too low", b.verify_confidence
154        );
155        println!(
156            "[bench] {} | {}→{} tokens | {:.1}% reduction | confidence {:.2}",
157            b.content_type, b.original_tokens, b.compressed_tokens,
158            b.reduction_pct, b.verify_confidence
159        );
160    }

162    /// Large JSON array — should sample via schema extraction.
163    /// Gate: ≥20% reduction, confidence ≥0.7, "count" or item count retained.
164    #[test]
165    fn bench_large_json_array() {
166        let items: Vec<String> = (1..=20)
167            .map(|i| format!(r#"{{"id":{i},"name":"item{i}","value":{i}00,"active":true}}"#))
168            .collect();
169        let input = format!("[{}]", items.join(","));
170        let b = run_bench("large_json_array", &input, "id");
171        assert_gates(&b, 20.0, 0.6);
172    }
