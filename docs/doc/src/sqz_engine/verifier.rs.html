1/// Two-pass compression verifier.
2///
3/// After compression, the verifier checks that critical information was
4/// preserved. If confidence is below the threshold, it signals the caller
5/// to fall back to a safer (less aggressive) compression mode.
6///
7/// Checks performed:
8/// 1. Required JSON keys present (if original was JSON)
9/// 2. Numeric fields unchanged (no value corruption)
10/// 3. Error/warning lines retained (critical signal preservation)
11/// 4. Diff hunk headers present (if input was a git diff)
12/// 5. File paths preserved (no path truncation)
13/// 6. Minimum content retention (output not too short vs input)
14
15use crate::types::VerifyResult;
16
17/// Confidence threshold below which fallback is triggered.
18const FALLBACK_THRESHOLD: f64 = 0.6;
19
20pub struct Verifier;
21
22impl Verifier {
23    /// Run all invariant checks on `original` → `compressed`.
24    ///
25    /// Returns a `VerifyResult` with confidence score and check details.
26    /// If `result.confidence < FALLBACK_THRESHOLD`, the caller should
27    /// re-compress with a safer preset.
28    pub fn verify(original: &str, compressed: &str) -> VerifyResult {
29        let mut passed = Vec::new();
30        let mut failed = Vec::new();
31
32        // Check 1: Minimum content retention (output must be ≥ 10% of input length)
33        let retention = if original.is_empty() {
34            1.0
35        } else {
36            compressed.len() as f64 / original.len() as f64
37        };
38        if retention >= 0.10 {
39            passed.push("min_retention".to_string());
40        } else {
41            failed.push((
42                "min_retention".to_string(),
43                format!("output is only {:.1}% of input length", retention * 100.0),
44            ));
45        }
46
47        // Check 2: Error/warning lines retained
48        let error_lines: Vec<&str> = original
49            .lines()
50            .filter(|l| {
51                let lower = l.to_lowercase();
52                lower.contains("error:") || lower.contains("warning:") || lower.contains("fatal:")
53                    || lower.contains("panic:") || lower.contains("exception:")
54            })
55            .collect();
56        if error_lines.is_empty() {
57            passed.push("error_lines".to_string());
58        } else {
59            let missing: Vec<&str> = error_lines
60                .iter()
61                .filter(|&&line| !compressed.contains(line.trim()))
62                .copied()
63                .collect();
64            if missing.is_empty() {
65                passed.push("error_lines".to_string());
66            } else {
67                failed.push((
68                    "error_lines".to_string(),
69                    format!("{} error/warning line(s) missing from output", missing.len()),
70                ));
71            }
72        }
73
74        // Check 3: File paths preserved (lines containing / or \ with extension)
75        let path_lines: Vec<&str> = original
76            .lines()
77            .filter(|l| {
78                (l.contains('/') || l.contains('\\'))
79                    && l.chars().any(|c| c == '.')
80                    && l.len() < 200 // skip very long lines
81            })
82            .take(20) // only check first 20 path-like lines
83            .collect();
84        if path_lines.is_empty() {
85            passed.push("file_paths".to_string());
86        } else {
87            let missing_paths = path_lines
88                .iter()
89                .filter(|&&line| {
90                    // Extract the path-like token and check it's in the output
91                    let token = line.split_whitespace()
92                        .find(|t| t.contains('/') || t.contains('\\'))
93                        .unwrap_or("");
94                    !token.is_empty() && !compressed.contains(token)
95                })
96                .count();
97            if missing_paths == 0 {
98                passed.push("file_paths".to_string());
99            } else {
100                failed.push((
101                    "file_paths".to_string(),
102                    format!("{missing_paths} file path(s) missing from output"),
103                ));
104            }
105        }
106
107        // Check 4: JSON key preservation (if original is JSON)
108        // We check that the compressed output contains at least 50% of the
109        // original top-level keys. Intentionally stripped keys are expected to be absent.
110        let orig_trimmed = original.trim();
111        if orig_trimmed.starts_with('{') || orig_trimmed.starts_with('[') {
112            if let Ok(orig_val) = serde_json::from_str::<serde_json::Value>(orig_trimmed) {
113                let orig_keys = collect_top_level_keys(&orig_val);
114                if orig_keys.is_empty() {
115                    passed.push("json_keys".to_string());
116                } else {
117                    let present: usize = orig_keys
118                        .iter()
119                        .filter(|&&k| compressed.contains(k))
120                        .count();
121                    let retention_ratio = present as f64 / orig_keys.len() as f64;
122                    // Pass if at least 50% of original keys are present
123                    if retention_ratio >= 0.5 {
124                        passed.push("json_keys".to_string());
125                    } else {
126                        let missing: Vec<&str> = orig_keys
127                            .iter()
128                            .filter(|&&k| !compressed.contains(k))
129                            .copied()
130                            .collect();
131                        failed.push((
132                            "json_keys".to_string(),
133                            format!("only {:.0}% of JSON keys retained; missing: {:?}",
134                                retention_ratio * 100.0,
135                                &missing[..missing.len().min(5)]),
136                        ));
137                    }
138                }
139            } else {
140                passed.push("json_keys".to_string()); // not valid JSON, skip
141            }
142        } else {
143            passed.push("json_keys".to_string()); // not JSON, skip
144        }
145
146        // Check 5: Diff hunk headers preserved (if input is a git diff)
147        let hunk_headers: Vec<&str> = original
148            .lines()
149            .filter(|l| l.starts_with("@@"))
150            .collect();
151        if hunk_headers.is_empty() {
152            passed.push("diff_hunks".to_string());
153        } else {
154            let missing_hunks = hunk_headers
155                .iter()
156                .filter(|&&h| !compressed.contains(h))
157                .count();
158            if missing_hunks == 0 {
159                passed.push("diff_hunks".to_string());
160            } else {
161                failed.push((
162                    "diff_hunks".to_string(),
163                    format!("{missing_hunks} diff hunk header(s) missing"),
164                ));
165            }
166        }
167
168        // Check 6: Numeric values preserved (spot-check first 10 numbers)
169        let numbers: Vec<&str> = original
170            .split(|c: char| !c.is_ascii_digit() && c != '.' && c != '-')
171            .filter(|s| !s.is_empty() && s.len() >= 2 && s.parse::<f64>().is_ok())
172            .take(10)
173            .collect();
174        if numbers.is_empty() {
175            passed.push("numeric_values".to_string());
176        } else {
177            let missing_nums = numbers
178                .iter()
179                .filter(|&&n| !compressed.contains(n))
180                .count();
181            if missing_nums == 0 {
182                passed.push("numeric_values".to_string());
183            } else {
184                failed.push((
185                    "numeric_values".to_string(),
186                    format!("{missing_nums} numeric value(s) missing from output"),
187                ));
188            }
189        }
190
191        // Compute confidence: ratio of passed checks to total checks
192        let total = passed.len() + failed.len();
193        let confidence = if total == 0 {
194            1.0
195        } else {
196            passed.len() as f64 / total as f64
197        };
198
199        let fallback_triggered = confidence < FALLBACK_THRESHOLD;
200
201        VerifyResult {
202            passed: failed.is_empty(),
203            confidence,
204            checks_passed: passed,
205            checks_failed: failed,
206            fallback_triggered,
207        }
208    }
209
210    /// Check if a verify result warrants fallback to safer compression.
211    pub fn should_fallback(result: &VerifyResult) -> bool {
212        result.fallback_triggered
213    }
214}
215
216fn collect_top_level_keys(value: &serde_json::Value) -> Vec<&str> {
217    match value {
218        serde_json::Value::Object(map) => map.keys().map(|k| k.as_str()).collect(),
219        _ => vec![],
220    }
221}

231    #[test]
232    fn verify_identical_passes_all() {
233        let text = "error: something went wrong\nfile: src/main.rs\n";
234        let result = Verifier::verify(text, text);
235        assert!(result.passed);
236        assert!((result.confidence - 1.0).abs() < f64::EPSILON);
237        assert!(!result.fallback_triggered);
238    }
