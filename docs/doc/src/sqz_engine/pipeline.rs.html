22/// The compression pipeline orchestrator.
23///
24/// Runs content through a sequence of configurable stages (condense, strip
25/// nulls, collapse arrays, etc.), then applies post-processing: RLE
26/// compression, sliding-window dedup, entropy truncation, token pruning,
27/// dictionary compression, and TOON encoding for JSON.
28///
29/// Stages are sorted by priority — lower numbers run first. You can insert
30/// custom stages with [`insert_stage`](CompressionPipeline::insert_stage).
31pub struct CompressionPipeline {
32    stages: Vec<Box<dyn crate::stages::CompressionStage>>,
33    toon_encoder: ToonEncoder,
34    token_counter: TokenCounter,
35    token_pruner: TokenPruner,
36    dict_compressor: std::cell::RefCell<DictCompressor>,
37    entropy_truncator: EntropyTruncator,
38    #[allow(dead_code)]
39    prompt_cache_detector: PromptCacheDetector,
40}
41

43    /// Construct the pipeline from a preset, creating all built-in stages
44    /// sorted by priority, plus the token pruner, dictionary compressor,
45    /// and entropy truncator.
46    pub fn new(_preset: &Preset) -> Self {
47        let mut stages: Vec<Box<dyn crate::stages::CompressionStage>> = vec![
48            Box::new(AnsiStripper),
49            Box::new(KeepFieldsStage),
50            Box::new(StripFieldsStage),
51            Box::new(CondenseStage),
52            Box::new(GitDiffFoldStage),
53            Box::new(StripNullsStage),
54            Box::new(FlattenStage),
55            Box::new(TruncateStringsStage),
56            Box::new(CollapseArraysStage),
57            Box::new(CustomTransformsStage),
58        ];
59        stages.sort_by_key(|s| s.priority());
60
61        Self {
62            stages,
63            toon_encoder: ToonEncoder,
64            token_counter: TokenCounter::new(),
65            token_pruner: TokenPruner::new(),
66            dict_compressor: std::cell::RefCell::new(DictCompressor::new()),
67            entropy_truncator: EntropyTruncator::new(),
68            prompt_cache_detector: PromptCacheDetector,
69        }
70    }

72    /// Run content through all enabled stages, then apply:
73    /// 1. Dictionary compression + TOON encoding for JSON
74    /// 2. Token pruning for prose/plain text
75    /// 3. Entropy-weighted truncation for long content
76    pub fn compress(
77        &self,
78        input: &str,
79        _ctx: &SessionContext,
80        preset: &Preset,
81    ) -> Result<CompressedContent> {
82        let model_family = model_family_from_preset(preset);
83        let tokens_original = self.token_counter.count(input, &model_family);
84
85        let mut content = Content {
86            raw: input.to_owned(),
87            content_type: ContentType::PlainText,
88            metadata: crate::types::ContentMetadata {
89                source: None,
90                path: None,
91                language: None,
92            },
93            tokens_original,
94        };
95
96        let mut stages_applied: Vec<String> = Vec::new();
97
98        for stage in &self.stages {
99            let config = stage_config_from_preset(stage.name(), preset);
100            if config.enabled {
101                stage.process(&mut content, &config)?;
102                stages_applied.push(stage.name().to_owned());
103            }
104        }
105
106        // Post-stage processing: apply advanced compression techniques
107
108        let is_json = ToonEncoder::is_json(&content.raw);
109
110        // RLE: collapse repeated patterns in non-JSON content (generalizes condense)
111        // Only apply when content is long enough to benefit
112        if !is_json && content.raw.len() > 200 {
113            if let Ok(rle_result) = crate::rle_compressor::rle_compress(&content.raw, 3) {
114                if rle_result.runs_collapsed > 0 {
115                    // Safety check: verify critical markers are preserved
116                    let has_error = content.raw.contains("ERROR") || content.raw.contains("error:");
117                    let rle_has_error = rle_result.text.contains("ERROR") || rle_result.text.contains("error:");
118                    if !has_error || rle_has_error {
119                        content.raw = rle_result.text;
120                        stages_applied.push("rle".to_owned());
121                    }
122                }
123            }
124        }
125
126        // Sliding window dedup: catch repeated substrings across non-adjacent lines
127        if !is_json && content.raw.len() > 300 {
128            if let Ok(sw_result) = crate::rle_compressor::sliding_window_dedup(&content.raw, 4) {
129                if sw_result.dedup_count > 0 {
130                    // Safety check: verify critical markers are preserved
131                    let has_error = content.raw.contains("ERROR") || content.raw.contains("error:");
132                    let sw_has_error = sw_result.text.contains("ERROR") || sw_result.text.contains("error:");
133                    if !has_error || sw_has_error {
134                        content.raw = sw_result.text;
135                        stages_applied.push("sliding_window_dedup".to_owned());
136                    }
137                }
138            }
139        }
140
141        // Entropy-weighted truncation for long non-JSON content
142        if !is_json && content.raw.len() > 500 {
143            if let Ok(trunc_result) = self.entropy_truncator.truncate_string(&content.raw) {
144                if trunc_result.segments_dropped > 0 {
145                    content.raw = trunc_result.text;
146                    stages_applied.push("entropy_truncate".to_owned());
147                }
148            }
149        }
150
151        // Token pruning for prose content (non-JSON, non-code)
152        if !is_json && content.raw.len() > 100 && looks_like_prose(&content.raw) {
153            if let Ok(prune_result) = self.token_pruner.prune(&content.raw) {
154                if prune_result.tokens_removed > 0 {
155                    content.raw = prune_result.text;
156                    stages_applied.push("token_prune".to_owned());
157                }
158            }
159        }
160
161        // Apply dictionary compression + TOON encoding for JSON
162        let data = if ToonEncoder::is_json(&content.raw) {
163            // Dictionary compression — observe and try to compress.
164            // Only use dict compression if it actually saves bytes (header overhead
165            // can make small payloads larger).
166            self.dict_compressor.borrow_mut().observe(&content.raw);
167            let dict_result = self.dict_compressor.borrow().compress(&content.raw)?;
168
169            if dict_result.substitutions > 0 && dict_result.bytes_saved > 50 {
170                // Dict compression was beneficial — encode the JSON body with TOON
171                stages_applied.push("dict_compress".to_owned());
172                if let Some(newline_pos) = dict_result.data.find("\n{") {
173                    let header = &dict_result.data[..newline_pos + 1];
174                    let json_body = &dict_result.data[newline_pos + 1..];
175                    if let Ok(json) = serde_json::from_str::<serde_json::Value>(json_body) {
176                        let encoded = self.toon_encoder.encode(&json)?;
177                        stages_applied.push("toon_encode".to_owned());
178                        format!("{header}{encoded}")
179                    } else {
180                        dict_result.data
181                    }
182                } else {
183                    dict_result.data
184                }
185            } else {
186                // No beneficial dict substitutions — standard TOON encoding
187                let json: serde_json::Value = serde_json::from_str(&content.raw)
188                    .map_err(|e| SqzError::Other(format!("pipeline: JSON parse error: {e}")))?;
189                let encoded = self.toon_encoder.encode(&json)?;
190                stages_applied.push("toon_encode".to_owned());
191                encoded
192            }
193        } else {
194            content.raw
195        };
196
197        let tokens_compressed = self.token_counter.count(&data, &model_family);
198        let compression_ratio = if tokens_original == 0 {
199            1.0
200        } else {
201            tokens_compressed as f64 / tokens_original as f64
202        };
203
204        Ok(CompressedContent {
205            data,
206            tokens_compressed,
207            tokens_original,
208            stages_applied,
209            compression_ratio,
210            provenance: crate::types::Provenance::default(),
211            verify: None,
212        })
213    }

215    /// Insert a plugin stage and re-sort by priority.
216    pub fn insert_stage(&mut self, stage: Box<dyn crate::stages::CompressionStage>) {
217        self.stages.push(stage);
218        self.stages.sort_by_key(|s| s.priority());
219    }

221    /// Rebuild stage list from a new preset (hot-reload support).
222    /// Built-in stages are recreated; plugin stages are dropped and must be
223    /// re-inserted by the caller.
224    pub fn reload_preset(&mut self, _preset: &Preset) -> Result<()> {
225        let mut stages: Vec<Box<dyn crate::stages::CompressionStage>> = vec![
226            Box::new(AnsiStripper),
227            Box::new(KeepFieldsStage),
228            Box::new(StripFieldsStage),
229            Box::new(CondenseStage),
230            Box::new(GitDiffFoldStage),
231            Box::new(StripNullsStage),
232            Box::new(FlattenStage),
233            Box::new(TruncateStringsStage),
234            Box::new(CollapseArraysStage),
235            Box::new(CustomTransformsStage),
236        ];
237        stages.sort_by_key(|s| s.priority());
238        self.stages = stages;
239        // Reset session-level state in dict compressor
240        self.dict_compressor.borrow_mut().reset();
241        Ok(())
242    }

245/// Derive the `ModelFamily` from the preset's model configuration.
246fn model_family_from_preset(preset: &Preset) -> ModelFamily {
247    match preset.model.family.to_lowercase().as_str() {
248        "anthropic" | "claude" => ModelFamily::AnthropicClaude,
249        "openai" | "gpt" => ModelFamily::OpenAiGpt,
250        "google" | "gemini" => ModelFamily::GoogleGemini,
251        other => ModelFamily::Local(other.to_string()),
252    }

255/// Build a `StageConfig` for a named stage from the preset's compression config.
256fn stage_config_from_preset(name: &str, preset: &Preset) -> StageConfig {
257    let c = &preset.compression;
258    match name {
259        "ansi_strip" => StageConfig {
260            enabled: true,
261            options: serde_json::Value::Object(Default::default()),
262        },
263        "keep_fields" => {
264            if let Some(cfg) = &c.keep_fields {
265                StageConfig {
266                    enabled: cfg.enabled,
267                    options: serde_json::json!({ "fields": cfg.fields }),
268                }
269            } else {
270                StageConfig::default()
271            }
272        }
273        "strip_fields" => {
274            if let Some(cfg) = &c.strip_fields {
275                StageConfig {
276                    enabled: cfg.enabled,
277                    options: serde_json::json!({ "fields": cfg.fields }),
278                }
279            } else {
280                StageConfig::default()
281            }
282        }
283        "condense" => {
284            if let Some(cfg) = &c.condense {
285                StageConfig {
286                    enabled: cfg.enabled,
287                    options: serde_json::json!({
288                        "max_repeated_lines": cfg.max_repeated_lines
289                    }),
290                }
291            } else {
292                StageConfig::default()
293            }
294        }
295        "git_diff_fold" => {
296            if let Some(cfg) = &c.git_diff_fold {
297                StageConfig {
298                    enabled: cfg.enabled,
299                    options: serde_json::json!({
300                        "max_context_lines": cfg.max_context_lines
301                    }),
302                }
303            } else {
304                // Default: enabled with 2 context lines
305                StageConfig {
306                    enabled: true,
307                    options: serde_json::json!({ "max_context_lines": 2 }),
308                }
309            }
310        }
311        "strip_nulls" => {
312            if let Some(cfg) = &c.strip_nulls {
313                StageConfig {
314                    enabled: cfg.enabled,
315                    options: serde_json::Value::Object(Default::default()),
316                }
317            } else {
318                StageConfig::default()
319            }
320        }
321        "flatten" => {
322            if let Some(cfg) = &c.flatten {
323                StageConfig {
324                    enabled: cfg.enabled,
325                    options: serde_json::json!({ "max_depth": cfg.max_depth }),
326                }
327            } else {
328                StageConfig::default()
329            }
330        }
331        "truncate_strings" => {
332            if let Some(cfg) = &c.truncate_strings {
333                StageConfig {
334                    enabled: cfg.enabled,
335                    options: serde_json::json!({ "max_length": cfg.max_length }),
336                }
337            } else {
338                StageConfig::default()
339            }
340        }
341        "collapse_arrays" => {
342            if let Some(cfg) = &c.collapse_arrays {
343                StageConfig {
344                    enabled: cfg.enabled,
345                    options: serde_json::json!({
346                        "max_items": cfg.max_items,
347                        "summary_template": cfg.summary_template
348                    }),
349                }
350            } else {
351                StageConfig::default()
352            }
353        }
354        "custom_transforms" => {
355            if let Some(cfg) = &c.custom_transforms {
356                StageConfig {
357                    enabled: cfg.enabled,
358                    options: serde_json::Value::Object(Default::default()),
359                }
360            } else {
361                StageConfig::default()
362            }
363        }
364        _ => StageConfig::default(),
365    }

368/// Heuristic: does this text look like prose (documentation, error messages,
369/// README content) rather than code or structured data?
370fn looks_like_prose(text: &str) -> bool {
371    let lines: Vec<&str> = text.lines().take(20).collect();
372    if lines.is_empty() {
373        return false;
374    }
375
376    let mut prose_lines = 0;
377    let mut code_lines = 0;
378
379    for line in &lines {
380        let trimmed = line.trim();
381        if trimmed.is_empty() {
382            continue;
383        }
384        // Code indicators
385        if trimmed.ends_with('{')
386            || trimmed.ends_with('}')
387            || trimmed.ends_with(';')
388            || trimmed.starts_with("fn ")
389            || trimmed.starts_with("pub ")
390            || trimmed.starts_with("let ")
391            || trimmed.starts_with("const ")
392            || trimmed.starts_with("import ")
393            || trimmed.starts_with("def ")
394            || trimmed.starts_with("class ")
395            || trimmed.contains("::")
396            || trimmed.contains("->")
397            || trimmed.contains("=>")
398        {
399            code_lines += 1;
400        } else {
401            prose_lines += 1;
402        }
403    }
404
405    // Consider it prose if more than half the lines look like prose
406    prose_lines > code_lines
407}

649    /// Recursive strategy that generates arbitrary serde_json::Value instances.
650    /// Mirrors the strategy in toon.rs tests.
651    fn arb_json_value() -> impl Strategy<Value = serde_json::Value> {
652        let leaf = prop_oneof![
653            Just(serde_json::Value::Null),
654            any::<bool>().prop_map(serde_json::Value::Bool),
655            any::<i64>().prop_map(|n| serde_json::json!(n)),
656            any::<f64>()
657                .prop_filter("must be finite", |f| f.is_finite())
658                .prop_map(|f| serde_json::json!(f)),
659            ".*".prop_map(serde_json::Value::String),
660        ];
661
662        leaf.prop_recursive(4, 64, 8, |inner| {
663            prop_oneof![
664                prop::collection::vec(inner.clone(), 0..8)

683            let preset = default_preset();
684            let pipeline = CompressionPipeline::new(&preset);
685

687            let result = pipeline.compress(&json_input, &ctx(), &preset)
688                .expect("compress should not fail");
689
690            for ch in result.data.chars() {
691                let cp = ch as u32;
692                let is_printable_ascii = cp >= 0x20 && cp <= 0x7E;
693                let is_standard_whitespace = cp == 0x09 || cp == 0x0A || cp == 0x0D;
694                prop_assert!(
695                    is_printable_ascii || is_standard_whitespace,
696                    "non-ASCII-safe character in output: U+{:04X} ({:?})\noutput: {:?}",
697                    cp, ch, result.data
698                );
699            }

703    proptest! {
704        /// **Validates: Requirements 1.3**
705        ///
706        /// Property 1: Compression preserves semantically significant content.
707        ///
708        /// For any CLI output containing significant tokens (errors, warnings,
709        /// status messages) mixed with noise (repeated identical lines), the
710        /// Compression_Pipeline SHALL produce output that:
711        ///   1. Contains all significant lines.
712        ///   2. Contains each noise line at most `max_repeated_lines` times.
713        #[test]
714        fn prop_compression_preserves_significant_content(
715            significant_lines in prop::collection::vec(significant_line_strategy(), 1..=5),
716            noise_line in noise_line_strategy(),
717            noise_repeat in 5u32..=10u32,
718        ) {
719            let preset = default_preset(); // condense enabled, max_repeated_lines=3
720            let pipeline = CompressionPipeline::new(&preset);
721
722            // Build interleaved input: noise, significant, noise, significant, ...
723            let mut lines: Vec<String> = Vec::new();
724            for sig in &significant_lines {
725                for _ in 0..noise_repeat {
726                    lines.push(noise_line.clone());
727                }
728                lines.push(sig.clone());
729            }
730            // Trailing noise block
731            for _ in 0..noise_repeat {
732                lines.push(noise_line.clone());
733            }
734
735            let input = lines.join("\n");
736            let result = pipeline.compress(&input, &ctx(), &preset).unwrap();
737            let output = &result.data;
738
739            // 1. All significant lines must appear in the output.
740            for sig in &significant_lines {
741                prop_assert!(
742                    output.contains(sig.as_str()),
743                    "significant line missing from output: {:?}\noutput: {:?}",
744                    sig,
745                    output
746                );
747            }
748
749            // 2. No consecutive run of the noise line exceeds max_repeated_lines (3).
750            let mut max_run = 0usize;
751            let mut current_run = 0usize;
752            for line in output.lines() {
753                if line == noise_line.as_str() {
754                    current_run += 1;
755                    max_run = max_run.max(current_run);
756                } else {
757                    current_run = 0;
758                }
759            }
760            prop_assert!(
761                max_run <= 3,
762                "noise line {:?} has a consecutive run of {} (max 3)\noutput: {:?}",
763                noise_line,
764                max_run,
765                output
766            );
767        }
768    }
