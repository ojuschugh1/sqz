1//! Compression Cascades — multi-level compression that degrades gracefully
2//! as content ages out of relevance.
3//!
4//! Level 0 (fresh):   Full compressed content
5//! Level 1 (aging):   Signatures + changed lines only
6//! Level 2 (old):     File name + public API summary
7//! Level 3 (ancient): One-line reference with metadata
8//!
9//! Each level is a lossy compression of the previous level, but sqz controls
10//! what's lost (unlike the LLM's compaction which is unpredictable).
11
12
13/// Compression cascade level.
14#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
15pub enum CascadeLevel {
16    /// Full compressed content (current behavior).
17    Fresh = 0,
18    /// Signatures + changed lines only.
19    Aging = 1,
20    /// File name + public API summary.
21    Old = 2,
22    /// One-line reference.
23    Ancient = 3,
24}
25
26impl CascadeLevel {
27    /// Determine the cascade level based on age (turns since last access).
28    pub fn from_age(age: u64, thresholds: &CascadeThresholds) -> Self {
29        if age < thresholds.aging {
30            CascadeLevel::Fresh
31        } else if age < thresholds.old {
32            CascadeLevel::Aging
33        } else if age < thresholds.ancient {
34            CascadeLevel::Old
35        } else {
36            CascadeLevel::Ancient
37        }
38    }
39}
40
41/// Configurable thresholds for cascade levels (in turns).
42#[derive(Debug, Clone)]
43pub struct CascadeThresholds {
44    /// Turns before content moves from Fresh to Aging.
45    pub aging: u64,
46    /// Turns before content moves from Aging to Old.
47    pub old: u64,
48    /// Turns before content moves from Old to Ancient.
49    pub ancient: u64,
50}
51
52impl Default for CascadeThresholds {
53    fn default() -> Self {
54        Self {
55            aging: 10,
56            old: 20,
57            ancient: 30,
58        }
59    }
60}

62/// Result of cascade compression.
63#[derive(Debug, Clone)]
64pub struct CascadeResult {
65    pub text: String,
66    pub level: CascadeLevel,
67    pub tokens_original: u32,
68    pub tokens_compressed: u32,
69}
70
71/// Apply cascade compression to content based on its age.
72///
73/// - Fresh: return content as-is (already compressed by the pipeline)
74/// - Aging: extract function/class signatures + any changed lines
75/// - Old: file name + public API count
76/// - Ancient: one-line reference
77pub fn cascade_compress(
78    content: &str,
79    file_path: &str,
80    age: u64,
81    thresholds: &CascadeThresholds,
82) -> CascadeResult {
83    let level = CascadeLevel::from_age(age, thresholds);
84    let tokens_original = approx_tokens(content);
85
86    let text = match level {
87        CascadeLevel::Fresh => content.to_string(),
88        CascadeLevel::Aging => compress_aging(content, file_path),
89        CascadeLevel::Old => compress_old(content, file_path),
90        CascadeLevel::Ancient => compress_ancient(content, file_path),
91    };
92
93    let tokens_compressed = approx_tokens(&text);
94
95    CascadeResult {
96        text,
97        level,
98        tokens_original,
99        tokens_compressed,
100    }
101}

103/// Level 1: Extract signatures and key lines.
104fn compress_aging(content: &str, file_path: &str) -> String {
105    let mut result = Vec::new();
106    result.push(format!("// {file_path} (signatures)"));
107
108    for line in content.lines() {
109        let trimmed = line.trim();
110        // Keep function/class/struct/impl signatures
111        if trimmed.starts_with("pub fn ")
112            || trimmed.starts_with("fn ")
113            || trimmed.starts_with("pub struct ")
114            || trimmed.starts_with("struct ")
115            || trimmed.starts_with("pub enum ")
116            || trimmed.starts_with("impl ")
117            || trimmed.starts_with("pub trait ")
118            || trimmed.starts_with("class ")
119            || trimmed.starts_with("def ")
120            || trimmed.starts_with("function ")
121            || trimmed.starts_with("export ")
122            || trimmed.starts_with("import ")
123            || trimmed.starts_with("use ")
124        {
125            result.push(trimmed.to_string());
126        }
127        // Keep error/warning lines
128        if trimmed.contains("error") || trimmed.contains("Error")
129            || trimmed.contains("TODO") || trimmed.contains("FIXME")
130        {
131            result.push(trimmed.to_string());
132        }
133    }
134
135    if result.len() <= 1 {
136        // No signatures found — keep first and last 3 lines
137        let lines: Vec<&str> = content.lines().collect();
138        result.push("// (no signatures detected, showing head/tail)".into());
139        for line in lines.iter().take(3) {
140            result.push(line.to_string());
141        }
142        if lines.len() > 6 {
143            result.push(format!("// ... {} lines omitted ...", lines.len() - 6));
144        }
145        for line in lines.iter().rev().take(3).rev() {
146            result.push(line.to_string());
147        }
148    }
149
150    result.join("\n")
151}

153/// Level 2: File name + public API count.
154fn compress_old(content: &str, file_path: &str) -> String {
155    let pub_count = content.lines()
156        .filter(|l| {
157            let t = l.trim();
158            t.starts_with("pub fn ") || t.starts_with("pub struct ")
159                || t.starts_with("pub enum ") || t.starts_with("pub trait ")
160                || t.starts_with("export ") || t.starts_with("def ")
161        })
162        .count();
163
164    let line_count = content.lines().count();
165    let token_count = approx_tokens(content);
166
167    format!(
168        "[{file_path}: {line_count} lines, {pub_count} public items, ~{token_count} tokens]"
169    )
170}
171
172/// Level 3: One-line reference.
173fn compress_ancient(_content: &str, file_path: &str) -> String {
174    format!("[{file_path}]")
175}
176
177fn approx_tokens(s: &str) -> u32 {
178    ((s.len() as f64) / 4.0).ceil() as u32
179}

181#[cfg(test)]
182mod tests {
183    use super::*;
184
185    const SAMPLE_CODE: &str = r#"use std::collections::HashMap;
186
187pub struct Config {
188    pub name: String,
189    pub value: i32,
190}
191
192impl Config {
193    pub fn new(name: &str, value: i32) -> Self {
194        Self { name: name.to_string(), value }
195    }
196
197    pub fn validate(&self) -> bool {
198        !self.name.is_empty() && self.value > 0
199    }
200}
201
202fn helper(x: i32) -> i32 {
203    x * 2
204}
205"#;
206
207    #[test]
208    fn test_fresh_returns_unchanged() {
209        let result = cascade_compress(SAMPLE_CODE, "config.rs", 0, &CascadeThresholds::default());
210        assert_eq!(result.level, CascadeLevel::Fresh);
211        assert_eq!(result.text, SAMPLE_CODE);
212    }
213
214    #[test]
215    fn test_aging_extracts_signatures() {
216        let result = cascade_compress(SAMPLE_CODE, "config.rs", 15, &CascadeThresholds::default());
217        assert_eq!(result.level, CascadeLevel::Aging);
218        assert!(result.text.contains("pub struct Config"));
219        assert!(result.text.contains("pub fn new"));
220        assert!(result.text.contains("pub fn validate"));
221        assert!(result.tokens_compressed < result.tokens_original);
222    }
223
224    #[test]
225    fn test_old_shows_summary() {
226        let result = cascade_compress(SAMPLE_CODE, "config.rs", 25, &CascadeThresholds::default());
227        assert_eq!(result.level, CascadeLevel::Old);
228        assert!(result.text.contains("config.rs"));
229        assert!(result.text.contains("public items"));
230        assert!(result.tokens_compressed < result.tokens_original);
231    }
232
233    #[test]
234    fn test_ancient_one_line() {
235        let result = cascade_compress(SAMPLE_CODE, "config.rs", 35, &CascadeThresholds::default());
236        assert_eq!(result.level, CascadeLevel::Ancient);
237        assert_eq!(result.text, "[config.rs]");
238        assert!(result.tokens_compressed <= 5);
239    }
240
241    #[test]
242    fn test_custom_thresholds() {
243        let thresholds = CascadeThresholds {
244            aging: 5,
245            old: 10,
246            ancient: 15,
247        };
248        let result = cascade_compress(SAMPLE_CODE, "x.rs", 7, &thresholds);
249        assert_eq!(result.level, CascadeLevel::Aging);
250    }
251
252    #[test]
253    fn test_cascade_level_ordering() {
254        assert!(CascadeLevel::Fresh < CascadeLevel::Aging);
255        assert!(CascadeLevel::Aging < CascadeLevel::Old);
256        assert!(CascadeLevel::Old < CascadeLevel::Ancient);
257    }
258
259    #[test]
260    fn test_empty_content() {
261        let result = cascade_compress("", "empty.rs", 15, &CascadeThresholds::default());
262        assert_eq!(result.level, CascadeLevel::Aging);
263        assert!(!result.text.is_empty()); // at least the header
264    }
265
266    #[test]
267    fn test_tokens_decrease_with_level() {
268        let thresholds = CascadeThresholds::default();
269        let fresh = cascade_compress(SAMPLE_CODE, "x.rs", 0, &thresholds);
270        let aging = cascade_compress(SAMPLE_CODE, "x.rs", 15, &thresholds);
271        let old = cascade_compress(SAMPLE_CODE, "x.rs", 25, &thresholds);
272        let ancient = cascade_compress(SAMPLE_CODE, "x.rs", 35, &thresholds);
273
274        assert!(aging.tokens_compressed <= fresh.tokens_compressed);
275        assert!(old.tokens_compressed <= aging.tokens_compressed);
276        assert!(ancient.tokens_compressed <= old.tokens_compressed);
277    }
