1/// Tee Mode — saves full uncompressed command output for later recovery.
2///
3/// Configurable as `always`, `failures` (non-zero exit), or `never` (default).
4/// Saved outputs are timestamped files in a configurable directory.
5///
6/// Requirements: 38.1, 38.2, 38.3, 38.4
7
8use crate::{Result, SqzError};
9use chrono::Utc;
10use serde::{Deserialize, Serialize};
11use std::path::{Path, PathBuf};

13// ── TeeMode enum ─────────────────────────────────────────────────────────
14
15/// Controls when uncompressed output is saved.
16#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
17#[serde(rename_all = "lowercase")]
18pub enum TeeMode {
19    /// Save every command's uncompressed output.
20    Always,
21    /// Save only when the command exits with a non-zero code.
22    Failures,
23    /// Disabled (default).
24    Never,
25}
26
27impl Default for TeeMode {
28    fn default() -> Self {
29        Self::Never
30    }
31}
32
33impl std::fmt::Display for TeeMode {
34    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
35        match self {
36            Self::Always => write!(f, "always"),
37            Self::Failures => write!(f, "failures"),
38            Self::Never => write!(f, "never"),
39        }
40    }
41}
42
43impl std::str::FromStr for TeeMode {
44    type Err = SqzError;
45    fn from_str(s: &str) -> Result<Self> {
46        match s.to_ascii_lowercase().as_str() {
47            "always" => Ok(Self::Always),
48            "failures" => Ok(Self::Failures),
49            "never" => Ok(Self::Never),
50            other => Err(SqzError::Other(format!(
51                "invalid tee mode '{other}': expected always, failures, or never"
52            ))),
53        }
54    }
55}

59/// Metadata for a single saved tee output.
60#[derive(Debug, Clone, Serialize, Deserialize)]
61pub struct TeeEntry {
62    /// Unique identifier (the filename stem, e.g. "20250715T103045Z_git-status").
63    pub id: String,
64    /// The command that produced the output.
65    pub command: String,
66    /// Process exit code.
67    pub exit_code: i32,
68    /// Timestamp when the output was saved (RFC 3339).
69    pub timestamp: String,
70    /// Byte length of the saved output.
71    pub size_bytes: u64,
72}

74// ── TeeManager ───────────────────────────────────────────────────────────
75
76/// Manages saving and retrieving uncompressed command outputs.
77pub struct TeeManager {
78    mode: TeeMode,
79    dir: PathBuf,
80}
81
82impl TeeManager {
83    /// Create a new `TeeManager`.
84    ///
85    /// * `mode` — when to save (`Always`, `Failures`, `Never`).
86    /// * `dir`  — directory for saved files (created on first write).
87    pub fn new(mode: TeeMode, dir: PathBuf) -> Self {
88        Self { mode, dir }
89    }
90
91    /// Create a `TeeManager` with the default directory (`~/.sqz/tee/`).
92    pub fn with_default_dir(mode: TeeMode) -> Self {
93        let home = std::env::var("HOME")
94            .or_else(|_| std::env::var("USERPROFILE"))
95            .map(PathBuf::from)
96            .unwrap_or_else(|_| PathBuf::from("."));
97        Self::new(mode, home.join(".sqz").join("tee"))
98    }
99
100    /// Return the active tee mode.
101    pub fn mode(&self) -> TeeMode {
102        self.mode
103    }
104
105    /// Return the output directory.
106    pub fn dir(&self) -> &Path {
107        &self.dir
108    }
109
110    /// Conditionally save uncompressed output based on the configured mode.
111    ///
112    /// Returns `Some(id)` when the output was saved, `None` when skipped.
113    pub fn save(&self, command: &str, output: &str, exit_code: i32) -> Result<Option<String>> {
114        let should_save = match self.mode {
115            TeeMode::Always => true,
116            TeeMode::Failures => exit_code != 0,
117            TeeMode::Never => false,
118        };
119
120        if !should_save {
121            return Ok(None);
122        }
123
124        // Ensure directory exists.
125        std::fs::create_dir_all(&self.dir)?;
126
127        let now = Utc::now();
128        let ts = now.format("%Y%m%dT%H%M%SZ").to_string();
129        let sanitised_cmd = sanitise_command(command);
130        let id = format!("{ts}_{sanitised_cmd}");
131
132        // Write the output file.
133        let output_path = self.dir.join(format!("{id}.txt"));
134        std::fs::write(&output_path, output)?;
135
136        // Write a small metadata sidecar.
137        let meta = TeeEntry {
138            id: id.clone(),
139            command: command.to_owned(),
140            exit_code,
141            timestamp: now.to_rfc3339(),
142            size_bytes: output.len() as u64,
143        };
144        let meta_path = self.dir.join(format!("{id}.json"));
145        let meta_json = serde_json::to_string_pretty(&meta)?;
146        std::fs::write(&meta_path, meta_json)?;
147
148        Ok(Some(id))
149    }
150
151    /// List all saved tee entries, most recent first.
152    pub fn list(&self) -> Result<Vec<TeeEntry>> {
153        if !self.dir.exists() {
154            return Ok(Vec::new());
155        }
156
157        let mut entries = Vec::new();
158        for entry in std::fs::read_dir(&self.dir)? {
159            let entry = entry?;
160            let path = entry.path();
161            if path.extension().and_then(|e| e.to_str()) == Some("json") {
162                match std::fs::read_to_string(&path) {
163                    Ok(json) => {
164                        if let Ok(te) = serde_json::from_str::<TeeEntry>(&json) {
165                            entries.push(te);
166                        }
167                    }
168                    Err(_) => continue, // skip unreadable files
169                }
170            }
171        }
172
173        // Sort newest first (lexicographic on id works because of the timestamp prefix).
174        entries.sort_by(|a, b| b.id.cmp(&a.id));
175        Ok(entries)
176    }
177
178    /// Retrieve the full saved output for a given id.
179    pub fn get(&self, id: &str) -> Result<String> {
180        let path = self.dir.join(format!("{id}.txt"));
181        if !path.exists() {
182            return Err(SqzError::Other(format!("tee entry '{id}' not found")));
183        }
184        Ok(std::fs::read_to_string(&path)?)
185    }
186}

190/// Turn a command string into a short, filesystem-safe slug.
191fn sanitise_command(cmd: &str) -> String {
192    let base = cmd
193        .split_whitespace()
194        .take(3)
195        .collect::<Vec<_>>()
196        .join("-");
197    base.chars()
198        .map(|c| if c.is_ascii_alphanumeric() || c == '-' { c } else { '_' })
199        .take(60)
200        .collect()
201}
